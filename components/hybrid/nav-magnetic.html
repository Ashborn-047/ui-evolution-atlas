Glass frosted container with magnetic link items that pull toward the cursor.

const links = document.querySelectorAll('.nav-mag-link');

links.forEach(link => {
    link.addEventListener('mousemove', (e) => {
        const rect = link.getBoundingClientRect();
        const x = (e.clientX - rect.left - rect.width/2) * 0.3;
        const y = (e.clientY - rect.top - rect.height/2) * 0.3;
        
        gsap.to(link, { x, y, duration: 0.3 });
    });
    
    link.addEventListener('mouseleave', () => {
        gsap.to(link, { x: 0, y: 0, duration: 0.5, ease: "elastic.out(1, 0.5)" });
    });
});
