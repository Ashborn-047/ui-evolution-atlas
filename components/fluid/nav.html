GSAP Animation

function moveBlob(el) {
    // Update active state
    document.querySelectorAll('.liquid-nav-item').forEach(i => i.classList.remove('active'));
    el.classList.add('active');
    
    const blob = document.getElementById('blob');
    
    // Animate blob to new position
    gsap.to(blob, {
        left: el.offsetLeft,
        width: el.offsetWidth,
        duration: 0.6,
        ease: "elastic.out(1, 0.5)"  // Bouncy!
    });
}
