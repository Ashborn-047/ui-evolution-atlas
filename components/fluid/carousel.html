Squeeze Carousel

Simulating soft-body physics. Items compress and skew based on drag velocity - the faster you drag, the more they squash.

GSAP Draggable

const track = document.getElementById('squeezeTrack');
const items = track.querySelectorAll('.squeeze-item');

Draggable.create(track, {
    type: "x",
    bounds: { minX: -200, maxX: 0 },
    inertia: true,
    
    onDrag: function() {
        // Calculate velocity-based squeeze
        const velocity = this.getDirection() === "left" ? -1 : 1;
        const squeeze = Math.min(Math.abs(this.deltaX) / 10, 0.3);
        
        items.forEach(item => {
            gsap.to(item, {
                scaleX: 1 - squeeze * 0.5,
                skewX: velocity * squeeze * 15,
                duration: 0.1
            });
        });
    },
    
    onDragEnd: function() {
        // Elastic reset
        items.forEach(item => {
            gsap.to(item, {
                scaleX: 1,
                skewX: 0,
                duration: 0.8,
                ease: "elastic.out(1, 0.3)"
            });
        });
    }
});
