Elastic Card

This card mimics surface tension. When you interact, it stretches towards your cursor like a living membrane.

Elastic Surface

GSAP Animation

const card = document.getElementById('elasticCard');

card.addEventListener('mousemove', (e) => {
    const rect = card.getBoundingClientRect();
    const x = (e.clientX - rect.left - rect.width/2) / 20;
    const y = (e.clientY - rect.top - rect.height/2) / 20;
    
    gsap.to(card, {
        x: x,
        y: y,
        scaleX: 1 + Math.abs(x) * 0.01,
        scaleY: 1 + Math.abs(y) * 0.01,
        duration: 0.3
    });
});

card.addEventListener('mouseleave', () => {
    gsap.to(card, {
        x: 0, y: 0,
        scaleX: 1, scaleY: 1,
        duration: 0.8,
        ease: "elastic.out(1, 0.3)"
    });
});
