Magnetic Button

The interface meets you halfway. The button pulls itself towards the cursor as if magnetized.

GSAP Animation

const area = document.getElementById('magArea');
const btn = document.getElementById('magBtn');

area.addEventListener('mousemove', (e) => {
    const rect = area.getBoundingClientRect();
    const centerX = rect.left + rect.width / 2;
    const centerY = rect.top + rect.height / 2;
    
    // Calculate pull distance (capped)
    const deltaX = (e.clientX - centerX) * 0.4;
    const deltaY = (e.clientY - centerY) * 0.4;
    
    gsap.to(btn, {
        x: deltaX,
        y: deltaY,
        duration: 0.3,
        ease: "power2.out"
    });
});

area.addEventListener('mouseleave', () => {
    // Snap back with elastic
    gsap.to(btn, {
        x: 0,
        y: 0,
        duration: 0.8,
        ease: "elastic.out(1, 0.3)"
    });
});
