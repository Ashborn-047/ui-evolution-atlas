Click Carousel

Before touchscreens, carousels were rigid slideshows. Navigation required precise clicks - no swipe, no momentum.

IMG_01

JavaScript

let currentSlide = 1;
const totalSlides = 5;

function changeSlide(direction) {
    currentSlide += direction;
    
    // Wrap around
    if (currentSlide > totalSlides) currentSlide = 1;
    if (currentSlide < 1) currentSlide = totalSlides;
    
    // Update display - instant, no animation
    document.getElementById('clickSlide').textContent = 
        'IMG_0' + currentSlide;
}
